package me.moehritz.bcex;


import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;

public class SQLConnectionManager {

	private static Connection dbConnection = null;

	protected static Connection getConnection() {
		try {
			if ((dbConnection == null) || (dbConnection.isClosed())) {
				Class.forName("com.mysql.jdbc.Driver");
				dbConnection = DriverManager.getConnection(Settings.dbUrl,
						Settings.dbUsername, Settings.dbPassword);
			}
			return dbConnection;
		} catch (SQLException e) {
			BCEX.getInstance().error("Error getting SQL-connection: " + e.getMessage());
			e.printStackTrace();
			return null;
		} catch (ClassNotFoundException e) {
			BCEX.getInstance().error("Error getting SQL-connection: " + e.getMessage());
			e.printStackTrace();
		}
		return null;
	}

	protected static void closeConnection() {
		try {
			if ((dbConnection != null) && (!dbConnection.isClosed()))
				dbConnection.close();
		} catch (SQLException e) {
			BCEX.getInstance().error("Error closing SQL-connection: " + e.getMessage());
			e.printStackTrace();
		}
	}
}
